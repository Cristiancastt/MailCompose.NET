var email = Email.Compose(ctx =>
{
    ctx.Column(spacing: 20, () =>
    {
        ctx.Text("example").FontSerif().Text2Xl().TextColor(TailwindColors.Gray900)
            .LeadingTight().Center();

        ctx.Divider(TailwindColors.Gray900, thickness: 2, marginVertical: 8);

        ctx.Heading("Your subscription has been renewed", level: 2)
            .TextColor(TailwindColors.Gray900).Center().FontSerif().LeadingTight();

        ctx.Text("Hello John,").TextColor(TailwindColors.Gray700);

        ctx.Text("We confirm that your Professional Plan subscription has been successfully renewed for another year. Here are the details:")
            .TextColor(TailwindColors.Gray600).LeadingRelaxed();

        ctx.Container(() =>
        {
            ctx.Column(spacing: 10, () =>
            {
                ctx.Row(spacing: 0, () =>
                {
                    ctx.Text("Plan").TextSm().TextColor(TailwindColors.Gray500).W("40%");
                    ctx.Text("Professional").TextSm().Bold().TextColor(TailwindColors.Gray800)
                        .Align(TextAlign.Right).W("60%");
                });
                ctx.Divider(TailwindColors.Gray200);
                ctx.Row(spacing: 0, () =>
                {
                    ctx.Text("Period").TextSm().TextColor(TailwindColors.Gray500).W("40%");
                    ctx.Text("Mar 2026 — Mar 2027").TextSm().Bold().TextColor(TailwindColors.Gray800)
                        .Align(TextAlign.Right).W("60%");
                });
                ctx.Divider(TailwindColors.Gray200);
                ctx.Row(spacing: 0, () =>
                {
                    ctx.Text("Amount").TextSm().TextColor(TailwindColors.Gray500).W("40%");
                    ctx.Text("$149.00 / year").TextSm().Bold().TextColor(TailwindColors.Gray800)
                        .Align(TextAlign.Right).W("60%");
                });
                ctx.Divider(TailwindColors.Gray200);
                ctx.Row(spacing: 0, () =>
                {
                    ctx.Text("Next renewal").TextSm().TextColor(TailwindColors.Gray500).W("40%");
                    ctx.Text("March 4, 2027").TextSm().Bold().TextColor(TailwindColors.Gray800)
                        .Align(TextAlign.Right).W("60%");
                });
            });
        }).P(20).Border(1, TailwindColors.Gray200).RoundedLg();

        ctx.Text("If you have any questions about your subscription, don't hesitate to contact us.")
            .TextColor(TailwindColors.Gray600);

        ctx.Text("Best regards,").TextColor(TailwindColors.Gray600);
        ctx.Text("The Example Team").Bold().TextColor(TailwindColors.Gray800);

        ctx.Spacer(8);

        ctx.Row(spacing: 16, () =>
        {
            ctx.Link("https://example.com/account", "My Account")
                .TextColor(TailwindColors.Gray600).TextSm().Underline();
            ctx.Link("https://example.com/billing", "Billing")
                .TextColor(TailwindColors.Gray600).TextSm().Underline();
            ctx.Link("https://example.com/help", "Help Center")
                .TextColor(TailwindColors.Gray600).TextSm().Underline();
        });
    });
});

email.WithTheme(EmailTheme.Minimal);

email.WithTheme(t =>
{
    t.Title = "Subscription Renewed";
    t.BodyBackground = TailwindColors.White;
    t.ContentBackground = TailwindColors.White;
    t.ContentBorderRadius = 0;
    t.ContentPadding = 24;
    t.FooterText = null;
});

var html = email.Render();
