var email = Email.Compose(ctx =>
{
    ctx.Column(spacing: 16, () =>
    {
        ctx.Heading("Reset your password", level: 2)
            .TextColor(TailwindColors.Gray900);

        ctx.Text("We received a request to reset your account password.")
            .TextColor(TailwindColors.Gray600);

        ctx.Text("Click the button below to create a new password:");

        ctx.Button("Reset Password", "https://example.com/reset?token=abc123")
            .Background(TailwindColors.Red600);

        ctx.Divider();

        ctx.Text("If you did not request this, ignore this email. Your password will remain unchanged.")
            .FontSize(13)
            .TextColor(TailwindColors.Gray400)
            .Italic();
    });
});

email.WithTheme(t => t.Title = "Password Reset");
var html = email.Render();
